SPARQL quick reference


      
All triples (sample)

      
SELECT ?s ?p ?o WHERE { ?s ?p ?o } LIMIT 10


      
Match by type

      
PREFIX ex: <http://example.org/>
SELECT ?x WHERE { ?x a ex:Person }


      
Filter by value

      
PREFIX ex: <http://example.org/>
SELECT ?person ?age WHERE {
    ?person a ex:Person ; ex:age ?age .
    FILTER (?age < 18)
}


      
Optional data (like LEFT JOIN)

      
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
SELECT ?s ?label WHERE {
    ?s ?p ?o .
    OPTIONAL { ?s rdfs:label ?label }
}


      
Follow a relationship

      
PREFIX ex: <http://example.org/>
SELECT ?contact WHERE {
    ex:Alice ex:knows ?contact .
}


      
Named graph

      
SELECT ?s ?p ?o WHERE {
    GRAPH <http://example.org/myGraph> { ?s ?p ?o }
}


      

        Full guide: docs/user/sparql-guide.md
        · W3C SPARQL 1.1 spec
